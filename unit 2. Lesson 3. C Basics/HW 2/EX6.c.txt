#include <stdio.h>

int main(void) {

int num1 , sum=0 , counter=1 ;	
printf("Enter an interger:" );
fflush(stdin);
fflush(stdout);
scanf("%d", &num1);

for(;counter<=num1;counter++)
{
sum+=counter;
}

}
